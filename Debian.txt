#Enable on restart at multi-user.target
sudo systemctl daemon-reload
sudo systemctl enable --now voting-primary-tcp.service voting-secondary-tcp.service voting-gateway.service

#Process status
sudo systemctl status voting-gateway.service
sudo journalctl -u voting-gateway.service -f

#Stop services
sudo systemctl stop voting-gateway.service voting-secondary-tcp.service voting-primary-tcp.service

#Start services
sudo systemctl start voting-gateway.service voting-secondary-tcp.service voting-primary-tcp.service
