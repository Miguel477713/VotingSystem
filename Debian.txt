#Enable on restart at multi-user.target
sudo systemctl daemon-reload
sudo systemctl enable --now voting-tcp.service voting-failover.service

#Process status
sudo systemctl status voting-gateway.service
sudo journalctl -u voting-gateway.service -f

#Stop services
sudo systemctl stop voting-gateway.service voting-tcp.service voting-failover.service

#Start services
sudo systemctl start voting-gateway.service voting-tcp.service

#Manual assignment of IP to ethernet interface
sudo ip addr add 169.254.236.20/24 dev eth0  
sudo ip link set eth0 up

#Reset IP interface
sudo ip addr flush dev eth0

#List listening TCP/UDP ports
ss -tulnp
